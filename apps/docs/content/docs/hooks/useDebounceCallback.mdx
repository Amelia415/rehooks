# useDebounceCallback

A custom hook that debounces a callback function, executing it after the specified delay.

## Usage

```tsx title="Component.tsx"
import { useDebounceCallback } from "~/hooks/useDebounceCallback";
import { useState } from "react";

function Component() {
  const [inputValue, setInputValue] = useState("");

  const debouncedLog = useDebounceCallback((value: string) => {
    console.log("Debounced Input Value:", value);
  }, 500);

  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const { value } = e.target;
    setInputValue(value);
    debouncedLog(value);
  };

  return (
    <div>
      <h2>Debounced Callback Example</h2>
      <input
        type="text"
        value={inputValue}
        onChange={handleChange}
        placeholder="Type something..."
      />
    </div>
  );
}
```

## API

### useDebounceCallback

```ts
function useDebounceCallback<T extends (...args: any[]) => void>(
  callback: T,
  delay: number,
): (...args: Parameters<T>) => void;
```

#### Parameters

| Name     | Type     | Description                         |
| -------- | -------- | ----------------------------------- |
| callback | `T`      | The callback function to debounce.  |
| delay    | `number` | The debounce delay in milliseconds. |

#### Returns

| Name                | Type                               | Description                                   |
| ------------------- | ---------------------------------- | --------------------------------------------- |
| `debouncedCallback` | `(...args: Parameters<T>) => void` | A debounced version of the callback function. |
